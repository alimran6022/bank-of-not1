let mainBal = 0.0000; let fMine = 0.000000; let stMine = 0.000000; let slMine = 0.000000; let isStActive = false; let isSlActive = false; // মাইনিং ফাংশন যা প্রতি সেকেন্ডে আপডেট হবে function runMining() { setInterval(() => { // ১. ফ্রি মাইনিং - এটি অটোমেটিক চলবে fMine += 0.000002; let fEl = document.getElementById('free-counter'); if(fEl) { fEl.innerText = fMine.toFixed(6); } // ২. স্ট্যান্ডার্ড প্ল্যান (ইনভেস্ট করলে চলবে) let stEl = document.getElementById('standard-counter'); if(isStActive && stEl) { stMine += 0.000008; stEl.innerText = stMine.toFixed(6); } // ৩. সিলভার প্ল্যান (ইনভেস্ট করলে চলবে) let slEl = document.getElementById('silver-counter'); if(isSlActive && slEl) { slMine += 0.000015; slEl.innerText = slMine.toFixed(6); } }, 1000); } function invest(p) { if(confirm("Do you want to start this mining plan?")) { if(p === 'standard') isStActive = true; if(p === 'silver') isSlActive = true; alert("Success! Mining Started."); } } function claim(t) { if(t === 'free') { mainBal += fMine; fMine = 0; } else if(t === 'standard' && isStActive) { mainBal += stMine; stMine = 0; } else if(t === 'silver' && isSlActive) { mainBal += slMine; slMine = 0; } else { alert("Please invest first!"); return; } let bEl = document.getElementById('total-balance'); if(bEl) bEl.innerText = mainBal.toFixed(4) + " NOT"; alert("Claim Success!"); } // অ্যাপ লোড হওয়ার সাথে সাথে মাইনিং শুরু হবে window.onload = runMining;
